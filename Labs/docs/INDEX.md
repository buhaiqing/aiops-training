```
├── LEARNING_GUIDE.md            # 学习方法指南

│
├── lab1_xxx/
│   ├── README.md               # 快速指南（必需）
│   ├── lab_guide.md            # 详细指导（必需）
│   ├── generate_data.py        # 数据生成（必需）
│   ├── main.py                 # 主程序（必需）
│   ├── Makefile                # 自动化脚本（必需）
│   ├── requirements.txt        # 依赖说明（必需）
│   └── SUMMARY.md              # 完成总结（可选）
│
└── ... (其他 Lab 类似)
```
